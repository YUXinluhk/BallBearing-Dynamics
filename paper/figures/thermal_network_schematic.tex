\documentclass[border=10pt]{standalone}
\usepackage[siunitx, RPvoltages]{circuitikz}
\usepackage{amsmath, amssymb}
\usetikzlibrary{calc, positioning}

% Component scaling
\ctikzset{resistors/scale=0.7}
\ctikzset{capacitors/scale=0.6}
\ctikzset{sources/scale=0.7}

\begin{document}
\begin{tikzpicture}[
    every node/.style={font=\small},
    nodelabel/.style={font=\normalsize\bfseries, fill=white, inner sep=3pt},
    qlabel/.style={font=\footnotesize, text=red!70!black},
]

% ==================================================================
%    NODE POSITIONS
% ==================================================================
\coordinate (nIR)   at (0,   0);
\coordinate (nBall) at (7,   0);
\coordinate (nOR)   at (14,  0);
\coordinate (nOil)  at (7,  -6);

% Junction dots
\foreach \n in {nIR, nBall, nOR, nOil}
    \fill (\n) circle (2.5pt);

% Node labels — placed BELOW line to avoid resistor label collision
\node[nodelabel, below left=4pt and 2pt] at (nIR)   {$T_{\text{IR}}$};
\node[nodelabel, below left=4pt and 14pt] at (nBall) {$T_{\text{ball}}$};
\node[nodelabel, below left=4pt and 2pt] at (nOR)   {$T_{\text{OR}}$};
\node[nodelabel, below left=6pt and 10pt] at (nOil)  {$T_{\text{oil}}$};

% ==================================================================
%    THERMAL RESISTANCES
% ==================================================================
% R_IR-ball
\draw (nIR) to[R, l^=$R_{\text{IR-b}}$] (nBall);

% R_OR-ball
\draw (nBall) to[R, l^=$R_{\text{OR-b}}$] (nOR);

% R_ball-oil (vertical)
\draw (nBall) to[R, l_=$R_{\text{b-oil}}$] (nOil);

% R_OR-amb
\coordinate (nORamb) at ($(nOR) + (4, 0)$);
\draw (nOR) to[R, l^=$R_{\text{OR-amb}}$] (nORamb);
\draw (nORamb) node[ground]{};
\node[font=\footnotesize, above=5pt] at (nORamb) {$T_{\text{amb}}$};

% R_oil-amb
\coordinate (nOilamb) at ($(nOil) + (5.5, 0)$);
\draw (nOil) to[R, l^=$R_{\text{oil-amb}}$] (nOilamb);
\draw (nOilamb) node[ground]{};
\node[font=\footnotesize, above=5pt] at (nOilamb) {$T_{\text{amb}}$};

% ==================================================================
%    THERMAL CAPACITANCES
% ==================================================================
% C_IR (down from IR)
\coordinate (cIRbot) at ($(nIR) + (0, -2.5)$);
\draw (nIR) to[C, l_=$C_{\text{IR}}$] (cIRbot);
\draw (cIRbot) node[ground]{};

% C_ball (offset right to clear R_b-oil)
\coordinate (cBtop) at ($(nBall) + (2.2, 0)$);
\coordinate (cBbot) at ($(cBtop) + (0, -2.5)$);
\draw (nBall) -- (cBtop);
\draw (cBtop) to[C, l=$C_{\text{ball}}$] (cBbot);
\draw (cBbot) node[ground]{};

% C_OR (down from OR)
\coordinate (cORbot) at ($(nOR) + (0, -2.5)$);
\draw (nOR) to[C, l=$C_{\text{OR}}$] (cORbot);
\draw (cORbot) node[ground]{};

% C_oil (offset right to clear Q_conv)
\coordinate (cOtop) at ($(nOil) + (2.2, 0)$);
\coordinate (cObot) at ($(cOtop) + (0, -2.5)$);
\draw (nOil) -- (cOtop);
\draw (cOtop) to[C, l=$C_{\text{oil}}$] (cObot);
\draw (cObot) node[ground]{};

% ==================================================================
%    HEAT SOURCES (current sources — red)
% ==================================================================
% Q_IR: into Inner Race (from above)
\coordinate (qIR) at ($(nIR) + (0, 3.2)$);
\draw[red!70!black] (qIR) to[isource, color=red!70!black, l=$\dot{Q}_{\text{IR}}$] (nIR);
\node[qlabel, above=3pt, align=center] at (qIR)
    {\footnotesize Contact friction};

% Q_ball: into Ball (from above)
\coordinate (qBall) at ($(nBall) + (0, 3.2)$);
\draw[red!70!black] (qBall) to[isource, color=red!70!black, l_=$\dot{Q}_{\text{ball}}$] (nBall);
\node[qlabel, above=3pt, align=center] at (qBall)
    {\footnotesize Contact friction};

% Q_churn: into Oil (from the left)
\coordinate (qOil) at ($(nOil) + (-4.0, 0)$);
\draw[red!70!black] (qOil) to[isource, color=red!70!black, l_=$\dot{Q}_{\text{churn}}$] (nOil);
\node[qlabel, left=3pt, align=center] at (qOil)
    {\footnotesize Churning\\[-2pt]\footnotesize + Drag};

% ==================================================================
%    OIL CIRCULATION (convective cooling — teal, downward)
% ==================================================================
\coordinate (coolBot) at ($(nOil) + (0, -3.8)$);
\draw[teal!70!black] (nOil) to[isource, color=teal!70!black, l_=$\dot{Q}_{\text{conv}}$] (coolBot);
\node[font=\footnotesize, text=teal!70!black, below=3pt, align=center] at (coolBot)
    {$\dot{m}c_p\!\left(T_{\text{oil}} - T_{\text{in}}\right)$\\[-1pt]
     {\scriptsize Oil circulation}};

\end{tikzpicture}
\end{document}
